### BUG: Incorrect `outputPath` in `runBuild`
**Severity**: high
**Location**: `runBuild`
**Problem**: The `outputPath` is constructed using `path.join(buildDir, 'plugins', manifest.name)`. However, the `build-plugin.sh` script might output the plugin to a different location, or the `manifest.output` field (which is `./plugins/${config.name}`) might be relative to the `buildDir` but not necessarily directly inside a `plugins` subdirectory of `buildDir`. If the `build-plugin.sh` script or the manifest's `output` field specifies a different path, the `outputPath` calculated here will be incorrect, leading to `stats` being unable to find the plugin's contents and potentially incorrect paths being stored in the `PluginBuildResult`.
**Fix**: The `outputPath` should be derived from the `manifest.output` field, which explicitly states where the plugin will be built. Since `manifest.output` is defined as `./plugins/${config.name}`, the correct `outputPath` should be `path.join(buildDir, manifest.output)`.

/**
 * Run build-plugin.sh in the build directory and capture output.
 * Uses atomic replacement of the map entry to avoid torn reads.
 */
async function runBuild(buildId: string, buildDir: string, manifest: PluginManifest): Promise<void> {
  const existing = buildResults.get(buildId)
  if (!existing) return

  try {
    const output = await execPromise(
      path.join(buildDir, 'build-plugin.sh'),
      ['--clean'],
      { cwd: buildDir, timeout: 120000 }
    )

    // Parse output for stats.
    // Use manifest.name directly instead of parsing manifest.output's prefix, so the
    // path is always correct regardless of the exact format stored in manifest.output.
    const outputPath = path.join(buildDir, manifest.output); // FIX: Use manifest.output directly
    const stats = { skills: 0, scripts: 0, hooks: 0 }

    try {
      const skillEntries = await fs.readdir(path.join(outputPath, 'skills')).catch(() => [] as string[])
      stats.skills = skillEntries.length

      const scriptEntries = await fs.readdir(path.join(outputPath, 'scripts')).catch(() => [] as string[])
      stats.scripts = scriptEntries.length

      try {
        await fs.access(path.join(outputPath, 'hooks', 'hooks.json'))
        stats.hooks = 1
      } catch {
        stats.hooks = 0
      }
    } catch {
      // Stats collection failed — non-critical
    }

    // Atomic replacement: avoids torn reads from polling clients
    buildResults.set(buildId, {
      ...existing,
      status: 'complete',
      outputPath,
      logs: output.split('\n'),
      stats,
    })
  } catch (error: unknown) {
    const message = error instanceof Error ? error.message : String(error)
    const stderr = (error as any)?.stderr
    const logs = [message]
    if (stderr) logs.push(...String(stderr).split('\n'))

    // Atomic replacement
    buildResults.set(buildId, {
      ...existing,
      status: 'failed',
      logs,
    })
  }
}
