/**
 * Plugin Builder - Repo Scanner API
 *
 * POST /api/plugin-builder/scan-repo - Scan a git repo for skills
 */

import { NextResponse } from 'next/server'
import type { NextRequest } from 'next/server'
import { scanRepo } from '@/services/plugin-builder-service'

// Only GitHub HTTPS URLs are accepted to prevent SSRF attacks against internal hosts.
const GITHUB_URL_RE = /^https:\/\/github\.com\/[a-zA-Z0-9][a-zA-Z0-9._-]*\/[a-zA-Z0-9._-]+$/

export async function POST(request: NextRequest) {
  try {
    const body = await request.json()

    if (!body.url || typeof body.url !== 'string') {
      return NextResponse.json(
        { error: 'Repository URL is required' },
        { status: 400 }
      )
    }

    // Validate that the URL is a well-formed GitHub repository URL to prevent SSRF.
    if (!GITHUB_URL_RE.test(body.url)) {
      return NextResponse.json(
        { error: 'Invalid GitHub repository URL' },
        { status: 400 }
      )
    }

    // Guard against body.ref being a truthy non-string value (e.g. number, object)
    // which would bypass the falsy-check of `body.ref || 'main'`.
    const ref = typeof body.ref === 'string' ? body.ref : 'main'

    const result = await scanRepo(body.url, ref)

    if (result.error) {
      return NextResponse.json(
        { error: result.error },
        // Fall back to 500 when scanRepo omits the status field.
        { status: result.status ?? 500 }
      )
    }
    return NextResponse.json(result.data)
  } catch (error) {
    // SyntaxError is thrown by request.json() on malformed JSON — return 400.
    // All other errors (e.g. unexpected throws from scanRepo) are server faults — return 500.
    if (error instanceof SyntaxError) {
      console.error('Error parsing request body:', error)
      return NextResponse.json(
        { error: 'Invalid JSON in request body' },
        { status: 400 }
      )
    }
    console.error('Error scanning repo:', error)
    return NextResponse.json(
      { error: 'Internal server error' },
      { status: 500 }
    )
  }
}
