### BUG: `parseMultipart` `pos` update logic
**Severity**: low
**Location**: `parseMultipart`
**Problem**: In `parseMultipart`, the line `pos = contentStart` is intended to advance the position for the next boundary search. However, the next boundary is searched from `pos` (which is `contentStart`), but the boundary itself starts *before* `contentStart` (specifically, `boundaryStart` is where the boundary starts, and `partStart` is after the boundary and its `\r\n`). If `pos` is set to `contentStart`, it might miss a boundary that immediately follows the content without an intervening `\r\n` (though RFCs usually mandate `\r\n` before a boundary). A more robust approach might be to advance `pos` to `nextBoundary` or `contentEnd` to ensure the search for the *next* boundary starts after the *current* part has been fully processed.
**Fix**: Change `pos = contentStart` to `pos = contentEnd` to ensure the search for the next boundary starts after the current part's content.

```typescript
function parseMultipart(body: Buffer, contentType: string): { file: Buffer | null; options: string | null } {
  const boundaryMatch = contentType.match(/boundary=([^\s;]+)/)
  if (!boundaryMatch) return { file: null, options: null }

  // Boundary delimiter bytes as specified by RFC 2046
  const boundaryBuf = Buffer.from('\r\n--' + boundaryMatch[1])
  // The first boundary has no leading CRLF — prepend CRLF to body so the scan is uniform
  const normalised = Buffer.concat([Buffer.from('\r\n'), body])

  const CRLF_CRLF = Buffer.from('\r\n\r\n')

  let file: Buffer | null = null
  let options: string | null = null

  let pos = 0
  while (pos < normalised.length) {
    // Find next boundary
    const boundaryStart = normalised.indexOf(boundaryBuf, pos)
    if (boundaryStart === -1) break

    // Skip past the boundary bytes
    const afterBoundary = boundaryStart + boundaryBuf.length

    // Check for closing boundary (ends with '--')
    if (normalised[afterBoundary] === 0x2d && normalised[afterBoundary + 1] === 0x2d) break

    // Each part starts with \r\n after the boundary
    const partStart = afterBoundary + 2 // skip the \r\n that follows the boundary line
    if (partStart >= normalised.length) break

    // Find the end of the part headers (\r\n\r\n)
    const headerEnd = normalised.indexOf(CRLF_CRLF, partStart)
    if (headerEnd === -1) break

    // Part headers as UTF-8 string (ASCII-safe)
    const headers = normalised.slice(partStart, headerEnd).toString('utf8')

    // Content starts immediately after \r\n\r\n
    const contentStart = headerEnd + 4

    // Find where this part ends (next boundary position)
    const nextBoundary = normalised.indexOf(boundaryBuf, contentStart)
    const contentEnd = nextBoundary === -1 ? normalised.length : nextBoundary

    // Extract content as a raw Buffer slice — no string round-trip for binary data
    const contentBuf = normalised.slice(contentStart, contentEnd)

    if (headers.includes('name="file"')) {
      // Preserve exact bytes of binary file content (e.g. ZIP)
      file = contentBuf
    } else if (headers.includes('name="options"')) {
      // Text field — safe to decode as UTF-8
      options = contentBuf.toString('utf8')
    }

    pos = contentEnd // advance past current part's content end to find next boundary
  }

  return { file, options }
}
```
