export function getSkillKey(skill: PluginSkillSelection): string {
  switch (skill.type) {
    case 'core':
      return `core:${skill.name}`
    case 'marketplace':
      return `marketplace:${skill.id}`
    case 'repo':
      // Include ref so skills from different branches/tags have distinct keys

  }
}
