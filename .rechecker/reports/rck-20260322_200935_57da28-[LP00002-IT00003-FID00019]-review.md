### BUG: Concurrency counter `activeOps` not decremented on `runBuild` failure
**Severity**: high
**Location**: `buildPlugin`
**Problem**: The `runBuild` function is called in a fire-and-forget manner within `buildPlugin`. While `buildPlugin` has a `finally` block to decrement `activeOps`, this decrement happens when `buildPlugin` *returns*, not when the asynchronous `runBuild` operation *completes*. If `runBuild` fails (e.g., the build script times out or errors out), the `activeOps` counter is not decremented when `runBuild`'s promise rejects. This means that a failed build operation will permanently occupy one slot in the `MAX_CONCURRENT_OPS` limit, eventually leading to a denial of service for new build requests once `activeOps` reaches `MAX_CONCURRENT_OPS`.
**Fix**: The `runBuild` function should decrement `activeOps` in its own `finally` block to ensure the counter is released once the build operation (success or failure) is truly complete. The `buildPlugin`'s `finally` block should only decrement `activeOps` if `runBuild` was not initiated (i.e., if an error occurred *before* `runBuild` was called).

function validateGitUrl(url: string): string | null {
  if (!url || typeof url !== 'string') return 'URL is required'

  // Must be HTTPS
  if (!url.match(/^https:\/\/.+/)) {
    return 'Only HTTPS git URLs are allowed'
  }

  try {
    const parsed = new URL(url)
    const host = parsed.hostname.toLowerCase()

    // Block internal network addresses
    if (
      host === 'localhost' ||
      host === '127.0.0.1' ||
      host === '0.0.0.0' ||
      host === '::1' ||
      host.endsWith('.local') ||
      host.startsWith('10.') ||
      host.startsWith('192.168.') ||
      /^172\.(1[6-9]|2\d|3[01])\./.test(host)
    ) {
      return 'Internal network URLs are not allowed'
    }

    // Check against allowed hosts
    if (!ALLOWED_GIT_HOSTS.some(allowed => host === allowed || host.endsWith(`.${allowed}`))) {
      return `Git host "${host}" is not in the allowed list (${ALLOWED_GIT_HOSTS.join(', ')})`
    }

    return null // valid
  } catch {
    return 'Invalid URL format'
  }
}
