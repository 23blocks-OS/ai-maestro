/**
 * Plugin Builder - Repo Scanner API

 * POST /api/plugin-builder/scan-repo - Scan a git repo for skills

import { NextResponse } from 'next/server'
import type { NextRequest } from 'next/server'
import { scanRepo } from '@/services/plugin-builder-service'

export async function POST(request: NextRequest) {
  try {
    const body = await request.json()

    if (!body.url || typeof body.url !== 'string') {
      return NextResponse.json(
        { error: 'Repository URL is required' },
        { status: 400 }
      )
    }

    const result = await scanRepo(body.url, body.ref || 'main')

    // Assuming scanRepo always returns an object with 'error' and 'status' for error cases,
    // or 'data' for success. If scanRepo can throw, this needs a try-catch around it.
    if (result.error) {
      return NextResponse.json(
        { error: result.error },
        { status: result.status || 500 } // Provide a default status if scanRepo doesn't
      )
    }
    return NextResponse.json(result.data)
  } catch (error: any) { // Explicitly type error as 'any' or 'unknown' and narrow down
    console.error('Error scanning repo:', error)

    if (error instanceof SyntaxError) {
      // This error likely comes from request.json() failing to parse the body
      return NextResponse.json(
        { error: 'Invalid request body format' },
        { status: 400 }
      )
    } else {
      // Catch-all for other unexpected errors
      return NextResponse.json(
        { error: 'Internal server error' },
        { status: 500 }
      )
    }
  }
}
