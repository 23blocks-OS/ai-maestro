### BUG: `AgentRegistry.evictIfNeeded` infinite loop potential
**Severity**: high
**Location**: `AgentRegistry.evictIfNeeded`
**Problem**: The `evictIfNeeded` method has a potential infinite loop if all agents in `this.accessOrder` are currently in the `this.initializingAgents` map. In this scenario, `lruAgentId` is taken from `accessOrder`, checked against `initializingAgents`, and if found, it's put back at the end of `accessOrder`. If this happens for every agent in the `accessOrder` during a single `while` loop iteration, the `accessOrder` will effectively rotate without any agent being evicted, leading to an infinite loop if `this.agents.size >= this.maxAgents` remains true.
**Fix**: Introduce a mechanism to prevent re-checking the same agents repeatedly within a single call to `evictIfNeeded`. One way is to limit the number of eviction attempts or to mark agents that have been tried and failed to evict in the current pass. A simpler fix is to ensure that if an agent cannot be evicted (because it's initializing), we don't try to evict it again in the same pass.

```typescript
class AgentRegistry {
  // ...
  private async evictIfNeeded(): Promise<void> {
    const triedToEvict = new Set<string>(); // Keep track of agents we've tried to evict in this pass
    while (this.agents.size >= this.maxAgents && this.accessOrder.length > 0) {
      const lruAgentId = this.accessOrder.shift()!;

      if (triedToEvict.has(lruAgentId)) {
        // We've already tried this agent in this pass and couldn't evict it.
        // This means we're in a state where all LRU agents are initializing.
        // Break to avoid an infinite loop.
        this.accessOrder.unshift(lruAgentId); // Put it back at the front for next time
        break;
      }

      if (this.initializingAgents.has(lruAgentId)) {
        // Agent is initializing, cannot evict it yet. Put it back at the end.
        this.accessOrder.push(lruAgentId);
        triedToEvict.add(lruAgentId); // Mark as tried for this pass
        continue; // Try the next LRU agent
      }

      const agent = this.agents.get(lruAgentId);
      if (agent) {
        console.log(`[AgentRegistry] Evicting LRU agent ${lruAgentId.substring(0, 8)} (${this.agents.size}/${this.maxAgents})`);
        try {
          await agent.shutdown();
        } catch (err) {
          console.error(`[AgentRegistry] Error shutting down evicted agent:`, err);
        }
        this.agents.delete(lruAgentId);
        // Clear triedToEvict as we successfully evicted an agent,
        // and the state of accessOrder might have changed significantly.
        triedToEvict.clear();
      } else {
        // This case should ideally not happen if accessOrder and agents map are in sync.
        // If it does, it means an agent in accessOrder is not in agents map.
        // We should just continue to the next one.
        console.warn(`[AgentRegistry] Agent ${lruAgentId.substring(0, 8)} in accessOrder but not in agents map.`);
      }
    }
  }
```

```typescript
  // Store a tuple of [Promise<Agent>, AgentConfig, SubconsciousConfig]
  private initializingAgents = new Map<string, [Promise<Agent>, AgentConfig | undefined, SubconsciousConfig | undefined]>()
  private accessOrder: string[] = []  // Most recently accessed at the end
  private maxAgents: number

  // ... (constructor and other methods)

  async getAgent(agentId: string, config?: AgentConfig, subconsciousConfig?: SubconsciousConfig): Promise<Agent> {
    const existing = this.agents.get(agentId)
    if (existing) {
      this.touch(agentId)
      return existing
    }

    const inFlightEntry = this.initializingAgents.get(agentId)
    if (inFlightEntry) {
      const [inFlightPromise, inFlightConfig, inFlightSubconsciousConfig] = inFlightEntry;

      // A deep comparison might be needed for complex objects, but for simple objects, JSON.stringify is a quick check.
      if (JSON.stringify(config) !== JSON.stringify(inFlightConfig) || JSON.stringify(subconsciousConfig) !== JSON.stringify(inFlightSubconsciousConfig)) {

      }
      return inFlightPromise;
    }

    await this.evictIfNeeded()

    console.log(`[AgentRegistry] Loading agent ${agentId.substring(0, 8)} (${this.agents.size + 1}/${this.maxAgents})`)
    const agent = new Agent({
      agentId,
      workingDirectory: config?.workingDirectory

    const initPromise: Promise<Agent> = agent.initialize(subconsciousConfig).then(() => {

      this.touch(agentId)
      this.initializingAgents.delete(agentId)
      return agent
    }).catch(err => {
      this.initializingAgents.delete(agentId)
      throw err
    })

    this.initializingAgents.set(agentId, [initPromise, config, subconsciousConfig])
    return initPromise

    this.initializingAgents.delete(agentId) // This is correct

    }

  }
```
